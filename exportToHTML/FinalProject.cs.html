298      public class BookModel
299      {
300          private long id;
301          private string title;
302          private string writter;
303          private int price;
304  
305          public BookModel()
306          {
307          }
308  
309          public BookModel(long id, string title, string writter, int price)
310          {
311              this.id = id;
312              this.title = title;
313              this.writter = writter;
314              this.price = price;
315          }
316  
317          public long getID()
318          {
319              return id;
320          }
321  
322          public string getTitle()
323          {
324              return title;
325          }
326  
327          public string getWritter()
328          {
329              return writter;
330          }
331  
332          public int getPrice()
333          {
334              return price;
335          }
336  
337          public void setID(long id)
338          {
339              this.id = id;
340          }
341  
342          public void setTitle(string title)
343          {
344              this.title = title;
345          }
346  
347          public void setWritter(string writter)
348          {
349              this.writter = writter;
350          }
351  
352          public void setPrice(int price)
353          {
354              this.price = price;
355          }
356  
357          public override string ToString()
358          {
359              return "Book :" + "\n" + "-ID : " + id + "\n" + "-Title : " + title + "\n" + "-Writter : " + writter + "\n" + "-Price : " + price;
360          }
361      }
